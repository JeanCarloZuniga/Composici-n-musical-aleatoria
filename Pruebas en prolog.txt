dynamic progresion/1.

assert(progresion(a, (a,a,a))).
assert(progresion(a, (b,b,b))).
assert(progresion(a, (c,c,c))).
assert(progresion(a, (d,d,d))).

	tamano(a, X).    Devolvería 4.

	resultado_especifico(a, 3, X).    Devolvería (c,c,c).
